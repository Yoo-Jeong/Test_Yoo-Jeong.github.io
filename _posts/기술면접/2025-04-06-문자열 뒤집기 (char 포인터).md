```cpp
#include <iostream>
#include <cstring> // strlen 사용을 위해 필요

void reverse(char* str) {
    if (str == nullptr) return;

    size_t len = strlen(str);
    size_t left = 0;
    size_t right = len - 1;

    while (left < right) {
        // 세 번째 변수(temp)를 이용한 수동 교환
        char temp = str[left];
        str[left] = str[right];
        str[right] = temp;

        left++;
        right--;
    }
}
```

## swap 함수 사용

```cpp
#include <iostream>
#include <cstring> // strlen 사용을 위해 필요

void reverse(char* str) {
    if (str == nullptr) return;  // nullptr 방어

    size_t len = strlen(str);
    size_t left = 0;
    size_t right = len - 1;

    while (left < right) {
        std::swap(str[left], str[right]);
        left++;
        right--;
    }
}
```

### 테스트 코드
```cpp
int main() {
    char text[] = "hello";
    reverse(text);
    std::cout << text << std::endl;  // 출력: "olleh"
    return 0;
}
```
